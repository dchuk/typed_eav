  # frozen_string_literal: true

  require "spec_helper"

  # Round-4 review: parent_scope axis correctness for the two-level scope
  # partition introduced in Phase 1.
  #
  # Scenario A — Cross-tenant + cross-workspace under unscoped: a query in
  #   `TypedEAV.unscoped { }` must OR-across all field_ids that share a name,
  #   regardless of (scope, parent_scope) combination, while AND-ing across
  #   filters. The multimap branch is structurally unchanged from round_3 —
  #   this scenario verifies the parent_scope dimension is included in the
  #   "all fields for entity_type" sweep.
  #
  # Scenario B — Three-way collision precedence: when global, scope-only,
  #   and full-triple fields share a name on a (scope, parent_scope)-bearing
  #   entity, the full-triple wins. When only global + scope-only exist (no
  #   full-triple), scope-only wins. Mirrors round_3's two-way precedence
  #   extended to three dimensions.
  #
  # Scenario C — Orphan-parent rejection at Field level: a Field row with
  #   parent_scope set and scope nil is invalid. The model-level invariant
  #   (Field::Base#validate_parent_scope_invariant) prevents the row from
  #   ever existing in the DB.
  #
  # Scenario D — Cross-(scope, parent_scope) Value rejection: a Value
  #   attached to a Field whose parent_scope doesn't match the entity's
  #   typed_eav_parent_scope is invalid. Mirrors round_2's cross-scope guard
  #   extended to the parent_scope axis.
  RSpec.describe "Round-4 review: parent_scope partition", :scoping do
    before do
      TypedEAV.config.scope_resolver = nil
      TypedEAV.config.require_scope = true
    end

    after { TypedEAV.config.reset! }

    describe "Scenario A: unscoped + multi-tenant + multi-workspace" do
      # Three (tenant, workspace) partitions, each with its own status field.
      let!(:status_t1_w1) { create(:text_field, name: "status", entity_type: "Project", scope: "t1", parent_scope: "w1") }
      let!(:status_t1_w2) { create(:text_field, name: "status", entity_type: "Project", scope: "t1", parent_scope: "w2") }
      let!(:status_t2_w1) { create(:text_field, name: "status", entity_type: "Project", scope: "t2", parent_scope: "w1") }

      let!(:p_t1_w1) { create(:project, tenant_id: "t1", workspace_id: "w1").tap { |p| TypedEAV::Value.create!(entity: p, field: status_t1_w1, value: "active") } }
      let!(:p_t1_w2) { create(:project, tenant_id: "t1", workspace_id: "w2").tap { |p| TypedEAV::Value.create!(entity: p, field: status_t1_w2, value: "active") } }
      let!(:p_t2_w1) { create(:project, tenant_id: "t2", workspace_id: "w1").tap { |p| TypedEAV::Value.create!(entity: p, field: status_t2_w1, value: "active") } }

      let!(:p_t1_w1_inactive) { create(:project, tenant_id: "t1", workspace_id: "w1").tap { |p| TypedEAV::Value.create!(entity: p, field: status_t1_w1, value: "inactive") } }

      it "OR-collapses matches across all (scope, parent_scope) combinations under unscoped" do
        results = TypedEAV.unscoped do
          Project.where_typed_eav({ name: "status", op: :eq, value: "active" })
        end
        expect(results).to include(p_t1_w1, p_t1_w2, p_t2_w1)
        expect(results).not_to include(p_t1_w1_inactive)
      end
    end

    describe "Scenario B: three-way collision precedence" do
      let!(:global_age)     { create(:integer_field, name: "age", entity_type: "Project") }
      let!(:scope_only_age) { create(:integer_field, name: "age", entity_type: "Project", scope: "t1") }
      let!(:full_triple_age) { create(:integer_field, name: "age", entity_type: "Project", scope: "t1", parent_scope: "w1") }

      def captured_field
        captured = nil
        allow(TypedEAV::QueryBuilder).to receive(:entity_ids) do |field, _op, _value|
          captured = field
          TypedEAV::Value.none.select(:entity_id)
        end
        yield
        captured
      end

      it "full-triple wins on a (scope, parent_scope)-bearing entity" do
        picked = captured_field do
          TypedEAV.with_scope(["t1", "w1"]) do
            Project.where_typed_eav({ name: "age", op: :eq, value: 30 })
          end
        end
        expect(picked).to eq(full_triple_age)
      end

      it "scope-only wins when no full-triple field exists" do
        full_triple_age.destroy
        picked = captured_field do
          TypedEAV.with_scope(["t1", "w1"]) do
            Project.where_typed_eav({ name: "age", op: :eq, value: 30 })
          end
        end
        expect(picked).to eq(scope_only_age)
      end

      it "global wins when neither scoped variant exists" do
        full_triple_age.destroy
        scope_only_age.destroy
        picked = captured_field do
          TypedEAV.with_scope(["t1", "w1"]) do
            Project.where_typed_eav({ name: "age", op: :eq, value: 30 })
          end
        end
        expect(picked).to eq(global_age)
      end
    end

    describe "Scenario C: orphan-parent rejection at Field level" do
      it "rejects parent_scope.present? && scope.blank? at validation" do
        f = build(:integer_field, name: "x", entity_type: "Project", scope: nil, parent_scope: "w1")
        expect(f).not_to be_valid
        expect(f.errors[:parent_scope]).to be_present
      end

      it "rejects via Section as well (symmetric inline-duplicated guard)" do
        s = TypedEAV::Section.new(name: "S", code: "x", entity_type: "Project", scope: nil, parent_scope: "w1")
        expect(s).not_to be_valid
        expect(s.errors[:parent_scope]).to be_present
      end
    end

    describe "Scenario D: cross-(scope, parent_scope) Value rejection" do
      let!(:full_triple_field) { create(:integer_field, name: "x", entity_type: "Project", scope: "t1", parent_scope: "w1") }
      let!(:project_w1) { create(:project, tenant_id: "t1", workspace_id: "w1") }
      let!(:project_w2) { create(:project, tenant_id: "t1", workspace_id: "w2") }

      it "rejects a Value bound to a project in a different workspace" do
        v = TypedEAV::Value.new(entity: project_w2, field: full_triple_field, value: 1)
        expect(v).not_to be_valid
      end

      it "accepts a Value bound to a project in the matching workspace" do
        v = TypedEAV::Value.new(entity: project_w1, field: full_triple_field, value: 1)
        expect(v).to be_valid
      end

      it "rejects a Value bound to an entity host without parent_scope_method (Contact)" do
        contact = create(:contact, tenant_id: "t1")
        v = TypedEAV::Value.new(entity: contact, field: full_triple_field, value: 1)
        expect(v).not_to be_valid
      end
    end
  end
