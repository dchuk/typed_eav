Storage precision is governed by `Field::Decimal`'s inherited `precision_scale` option (which Percentage does NOT expose via store_accessor — but it's still settable on the underlying decimal_value column type, governed by the migration's column definition). decimal_places defaults to 2 (e.g., 0.755 renders as "75.50%" with display_as: :percent).

# frozen_string_literal: true

module TypedEAV
  module Field
    # Two-cell field type: stores `{amount: BigDecimal, currency: String}`
    # across decimal_value (amount) + string_value (currency ISO 4217 code).
    #
    # Phase 5 contract:
    # - value_columns: [:decimal_value, :string_value] — propagates through
    #   versioning's snapshot loop (Phase 4) and value-change-event filter.
    # - operator_column: :currency_eq → :string_value; everything else →
    #   :decimal_value. Routed via QueryBuilder.filter (plan 05-01).
    # - read_value / write_value / apply_default_to: paired overrides
    #   that compose / unpack the {amount, currency} hash across the two
    #   physical columns.
    # - cast: Hash input only. Bare Numeric/String is invalid — explicit
    #   currency dimension is required at write time.
    #
    # Options:
    # - default_currency: String ISO 4217 code (e.g., "USD"). Used as the
    #   currency fallback when cast input has amount but no currency.
    # - allowed_currencies: Array<String> of ISO codes. When set,
    #   validate_typed_value enforces inclusion.
    class Currency < Base
      value_column :decimal_value

      operators :eq, :gt, :lt, :gteq, :lteq, :between,
                :currency_eq,
                :is_null, :is_not_null

      store_accessor :options, :default_currency, :allowed_currencies

      validates :default_currency, format: { with: /\A[A-Z]{3}\z/ }, allow_nil: true
      validate :allowed_currencies_format

      # Phase 5: two-cell shape — overrides ColumnMapping.value_columns.
      def self.value_columns
        [:decimal_value, :string_value]
      end

      # Phase 5: operator_column dispatch (plan 05-01 contract).
      # :currency_eq routes to string_value; every other supported operator
      # routes to decimal_value. is_null / is_not_null target decimal_value
      # (a Currency value is null when its amount is null).
      def self.operator_column(operator)
        operator == :currency_eq ? :string_value : :decimal_value
      end

      # Cast Hash input → [{amount: BigDecimal, currency: String}, false]
      # or [nil, false] for nil/blank, or [nil, true] for unparseable input.
      def cast(raw)
        return [nil, false] if raw.nil? || (raw.respond_to?(:empty?) && raw.empty?)
        return [nil, true] unless raw.is_a?(Hash)

        amount_raw   = raw[:amount]   || raw["amount"]
        currency_raw = raw[:currency] || raw["currency"]

        amount_bd = nil
        if amount_raw.present?
          amount_bd = BigDecimal(amount_raw.to_s, exception: false)
          return [nil, true] if amount_bd.nil?
        end

        currency_str = currency_raw.is_a?(String) ? currency_raw.upcase : nil
        currency_str ||= default_currency if amount_bd && default_currency.present?
        # Reject non-3-letter currency codes (validation also catches this on
        # save; cast catches it earlier so :invalid is set on the cast result).
        if currency_str && currency_str !~ /\A[A-Z]{3}\z/
          return [nil, true]
        end

        [{ amount: amount_bd, currency: currency_str }, false]
      end

      # Read both columns; compose hash; nil when both are nil.
      def read_value(value_record)
        amount   = value_record[:decimal_value]
        currency = value_record[:string_value]
        return nil if amount.nil? && currency.nil?

        { amount: amount, currency: currency }
      end

      # Unpack casted hash into both columns. casted is the first element
      # of cast's return — already validated to be Hash-or-nil here.
      def write_value(value_record, casted)
        if casted.nil?
          value_record[:decimal_value] = nil
          value_record[:string_value]  = nil
        else
          value_record[:decimal_value] = casted[:amount]
          value_record[:string_value]  = casted[:currency]
        end
      end

      # Apply field-level default. default_value is whatever was stored at
      # field-save time (a Hash for Currency); unpack into both columns.
      def apply_default_to(value_record)
        d = default_value
        return if d.nil?
        return unless d.is_a?(Hash)

        value_record[:decimal_value] = d[:amount]   || d["amount"]
        value_record[:string_value]  = d[:currency] || d["currency"]
      end

      # Co-population validation + allowed_currencies inclusion.
      def validate_typed_value(record, val)
        return if val.nil?
        unless val.is_a?(Hash) && val[:amount].present? && val[:currency].present?
          record.errors.add(:value, "must have both amount and currency")
          return
        end
        allowed = options_hash[:allowed_currencies]
        if allowed.present? && !Array(allowed).include?(val[:currency])
          record.errors.add(:value, :inclusion)
        end
      end

      private

      def allowed_currencies_format
        list = options_hash[:allowed_currencies]
        return if list.nil?
        unless list.is_a?(Array) && list.all? { |c| c.is_a?(String) && c =~ /\A[A-Z]{3}\z/ }
          errors.add(:allowed_currencies, "must be an Array of 3-letter uppercase ISO codes")
        end
      end
    end
  end
end
